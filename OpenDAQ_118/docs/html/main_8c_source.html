211 uint8_t cal_val_input_delay(void)
212 {
213  // delays for 5 sec, allows input of cal values, names and offsets
214  char command[5];// command string
215  uint8_t calmode = 0; //flag for calmode or not
216  PORTD |= (1<<5); //LED on
217  uart_puts_p(PSTR("# type cal (in 5 sec)\n"));
218  uint32_t timeout= 0;
219  uint32_t counts_to_timeout = 1300000;// 1.3m = ~5 sec
220  UCSRB |= (1 << RXEN);
221  UCSRB |= (1 << RXCIE);//uart input enabled//
222  while( !(UCSRA & (1 << RXC)) )// check for nothing in the rx buff
223  {
224  timeout++;
225  if (timeout>=counts_to_timeout)//and not timed out
226  {break;}
227  }
228  if (timeout<counts_to_timeout)
229  {
230  if (uart_getc() == 'c')// have to do this to prevent noise from causing cal mode
231  {
232  uart_putc('c');
233  read_line(command, 4);// read up to 3 chars
234  if(strcmp_P(command, PSTR("al")) == 0)// checks if "al" is there
235  {
236  uart_puts_p(PSTR("Ver: "));uart_putw_dec(VERSION);
237  uart_putc('\n');
238  calmode = 1;
239  }
240  }
241  }
242  PORTD &=~ (1<<5); //LED off
243  return calmode;

491 uint16_t timesample(struct sample_data *data)
492 {
493  //check how long to run a sample set
494  uint16_t sample_secs;
495 
496  //start 16 bit timer
497  sixteen = 0;//16 second counter
498  TIMSK |= (1<<TOIE1); //set Overflow Interrupt Enable
499  TCNT1 = 0x00; // set the timer to 0
500  TCCR1B |= ((1<<CS10)|(1<<CS12)); // prescale the timer to be clock source to /1024 prescale
501  sei();
502  sample_all_channels(REPORT, data);//run one sample loop
503  TCCR1B |= 0x00;//Stop timer, check value
504  //time in ms = sixteen*1000 +TCNT/4.09
505  sample_secs = TCNT1/4;
506  sample_secs = sample_secs + (sixteen*1000);//time in ms
507  sample_secs = sample_secs + 600;//add 600 ms
508  sample_secs = sample_secs/1000;//time in sec, remainder left off
509  sample_secs = sample_secs + 1;//the next second
510  for (uint8_t i= 0; i<LINESIZE ;i++){data->serialout[i] = '\0';}//init (blank) serialout
511  uart_putc('\n');
512  return sample_secs;
513 }
