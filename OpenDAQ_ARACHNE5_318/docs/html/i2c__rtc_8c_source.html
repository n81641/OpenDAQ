170 uint32_t bintime2unixtime(RTC *rtc)
171 {
172 // Convert the BCD time From DS1305/6 to an unsigned long UNIX timestamp
173 // of seconds since the epoch-- Jan. 1, 1970.
174 //
175 // As this app never needs to go back in time or specify arbitrary dates,
176 // simplifications can be done by simply calculating the whole-year
177 // seconds through an arbitrary year such as 2000, 2007, or 2008.
178 //
179 
180 uint32_t work; // accumulator of seconds
181 signed int mult; // working
182 
183 // Base seconds, 1970 through 2008
184  work = RTC_SEC_TILL_2009;
185 
186 // How many years since?
187  mult = rtc->year - RTC_YEAR_MIN;
188  if ((mult < 0) || (mult > (RTC_YEAR_MAX - RTC_YEAR_MIN + 1)))
189  {
190  mult = 0;
191  }
192 // Add the whole years since start of 2009
193  work += (unsigned long)mult * RTC_SEC_PER_YEAR;
194 
195 // Add a day for each leap year in the prior years
196  work += ((unsigned long)mult/4) * RTC_SEC_PER_DAY;
197 
198 // Add a day for this year, if leap year and past Feb.
199  if ((mult+1) % 4 == 0)
200  {
201  // Leap year
202  if (rtc->month > 2)
203  {
204  // Past February
205  work += RTC_SEC_PER_DAY;
206  }
207  }
208 
209 // Add the days up to this month. Use a table
210  mult = rtc->month;
211  if ((mult < 1) || (mult > 12))
212  {
213  mult = 1;
214  }
215 
216  work += ((unsigned long)rtc_days[mult-1]) * RTC_SEC_PER_DAY;
217 
218 // Add the days this month, up to today
219  mult = rtc->mday;
220  if ((mult < 1) || (mult > 31))
221  {
222  mult = 1;
223  }
224 
225  work += ((unsigned long)(mult-1)) * RTC_SEC_PER_DAY;
226 
227 // Add the seconds up to this hour
228  mult = rtc->hour;
229  if ((mult < 0) || (mult > 23))
230  {
231  mult = 1;
232  }
233 
234  work += ((unsigned long)(mult)) * RTC_SEC_PER_HOUR;
235 
236 // Add the seconds this hour
237  mult = (unsigned int)60 * (unsigned int)rtc->min + (unsigned int)rtc->sec;
238 
239  work += mult;
240 
241  return work;
242 }
