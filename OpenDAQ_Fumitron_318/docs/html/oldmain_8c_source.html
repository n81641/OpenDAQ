557  i2c_start_wait(address+I2C_WRITE); // set device address and write mode
558  i2c_write(mux1); // Transmit first (of 2) input configure byte:
559  i2c_write(mux2); //Transmit second (of 2) input configure byte

561  _delay_ms(80); // 80 ok for normal channels, not enough for temp
562  if (mux2 == 0xC0){_delay_ms(80);}//internal temp read needs more time

567  uint16_t ret2 = i2c_readAck();// for i2c
568  uint8_t ret3 = i2c_readAck();// for i2c
569  i2c_stop();
570  //bit checks here:
571  if (bitisset(ret1,7) && bitisset(ret1,6) )
572  {
573  ADC_volts = 0xffff;
574  return(2);//over fullscale +
575  }

577  {

579  }

583  //which is bits 5,4,3,2,1,0 of ret1 (ret1<<10);//ret1 left shifted 10 sopts
584  // and 7.....0 of ret2 (ret2<<2);//ret2 left shifted 2 sopts
